```verus
proof fn phi_1_target_loses_mapped_status(
    pre: PageAllocator, post: PageAllocator,
    target_ptr: PagePtr, pcid: Pcid, va: VAddr,
)
    requires
        pre.wf(),
        pre.mapped_pages_4k().contains(target_ptr),
        !pre.page_mappings(target_ptr).contains((pcid, va)),
        pre.page_mappings(target_ptr).len() + pre.page_io_mappings(target_ptr).len() < usize::MAX,
        post.mapped_pages_4k() =~= pre.mapped_pages_4k(),
        post.mapped_pages_2m() =~= pre.mapped_pages_2m(),
        post.mapped_pages_1g() =~= pre.mapped_pages_1g(),
    ensures
        !post.page_is_mapped(target_ptr),
{
}
```
REASON: Adding a mapping must not cause the target page to lose its mapped status

```verus
}
```
