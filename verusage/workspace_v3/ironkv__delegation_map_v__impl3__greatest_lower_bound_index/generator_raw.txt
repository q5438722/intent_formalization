proof fn phi_3_glb_ignores_later_keys<K: KeyTrait + VerusClone>(m: StrictlyOrderedMap<K>, iter: KeyIterator<K>)
    requires
        m.valid(),
        m@.contains_key(K::zero_spec()),
        m.keys@.len() > 1,
        iter == KeyIterator::new_spec(m.keys@[m.keys@.len() - 1]),

REASON: When iter equals the last key, the GLB should be that key itself, not the minimum key — this would mean the search never advances past index 0
