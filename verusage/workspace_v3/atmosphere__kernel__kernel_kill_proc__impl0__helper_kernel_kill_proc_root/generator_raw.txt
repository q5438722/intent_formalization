```verus
    old_proc_dom: Set<ProcPtr>,
    new_proc_dom: Set<ProcPtr>,
    proc_ptr: ProcPtr,
    other_ptr: ProcPtr,
)
    requires
        old_proc_dom.contains(proc_ptr),
        old_proc_dom.contains(other_ptr),
        proc_ptr != other_ptr,
        new_proc_dom =~= old_proc_dom.remove(proc_ptr),
    ensures
        !new_proc_dom.contains(other_ptr),
{
}
```
REASON: Killing one process should never remove a different process from the domain
