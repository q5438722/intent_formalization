    requires
        !t.is_empty(),
        t.finite(),
    ensures ({
        let (s1, r1) = pop_rid(t);
        let (s2, r2) = pop_rid(t);
        r1 == r2 && s1 =~= s2
    }),
{
    let _ = pop_rid(t);
    let _ = pop_rid(t);
}
```
REASON: `choose()` is deterministic in Verus specs — calling `pop_rid` twice on the same set always picks the same element, but the spec doesn't document which element is chosen

proof fn phi_3_pop_rid_singleton_empty(r: ReqId)
    ensures ({
        let t = set!{r};
        let (s, popped) = pop_rid(t);
        s.is_empty() && popped == r
    }),

    let t = set!{r};
    let (s, popped) = pop_rid(t);

}
```
REASON: Popping from a singleton set yields the empty set and the only element — tests the minimal case
