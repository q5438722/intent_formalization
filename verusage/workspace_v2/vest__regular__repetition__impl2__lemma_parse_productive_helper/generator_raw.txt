```verus
proof fn phi_4_parse_helper_monotone_consumed<C: SecureSpecCombinator>(c: &RepeatN<C>, s: Seq<u8>, n1: usize, n2: usize, m1: int, m2: int, vs1: Seq<C::Type>, vs2: Seq<C::Type>)
    requires
        c.requires(),
        c.0.is_productive(),
        n1 < n2,
        c.spec_parse_helper(s, n1) == Some((m1, vs1)),
        c.spec_parse_helper(s, n2) == Some((m2, vs2)),
    ensures
        m2 > m1,
    decreases n2,
{
    c.lemma_parse_productive_helper(s, n2);
    if n2 == n1 + 1 {
        c.0.lemma_parse_productive(s.skip(m1 as int));
    } else {
        if let Some((m_prev, vs_prev)) = c.spec_parse_helper(s, (n2 - 1) as usize) {
            phi_4_parse_helper_monotone_consumed(c, s, n1, (n2 - 1) as usize, m1, m_prev, vs1, vs_prev);
            c.0.lemma_parse_productive(s.skip(m_prev as int));
        }
    }
}
```
REASON: With a productive inner combinator, each additional repetition must consume strictly more bytes — if this weren't true, the parser could loop without advancing through the input
