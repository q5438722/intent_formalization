```verus
proof fn phi_4_parse_consumed_monotone<C: SecureSpecCombinator>(
    c: &RepeatN<C>, s: Seq<u8>,
    n1: usize, n2: usize,
    m1: int, m2: int,
    vs1: Seq<C::Type>, vs2: Seq<C::Type>,
)
    requires
        c.requires(),
        n1 <= n2,
        c.spec_parse_helper(s, n1) == Some((m1, vs1)),
        c.spec_parse_helper(s, n2) == Some((m2, vs2)),
    ensures
        m1 <= m2,
    decreases n2,
{
    if n1 == n2 {
    } else {
        if let Some((m_prev, vs_prev)) = c.spec_parse_helper(s, (n2 - 1) as usize) {
            phi_4_parse_consumed_monotone(c, s, n1, (n2 - 1) as usize, m1, m_prev, vs1, vs_prev);
            c.0.lemma_parse_length(s.skip(m_prev as int));
        }
    }
}
```
REASON: Consumed bytes should be monotonically non-decreasing as repetition count grows — if more repetitions consumed fewer bytes, the parser would regress through input
