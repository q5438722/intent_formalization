```verus
proof fn phi_2_schedule_spec_match_ignores_metadata_diff(controller_id: int, vrs: VReplicaSetView, s: ClusterState)
    requires
        s.controller_and_externals.contains_key(controller_id),
        s.scheduled_reconciles(controller_id).contains_key(vrs.object_ref()),
        Cluster::the_object_in_schedule_has_spec_and_uid_as(controller_id, vrs)(s),
    ensures
        ({
            let sched_obj = s.scheduled_reconciles(controller_id)[vrs.object_ref()];
            let sched_vrs = VReplicaSetView::unmarshal(sched_obj)->Ok_0;
            sched_vrs.spec() == vrs.spec()
            && sched_obj.metadata.uid == vrs.metadata().uid
            && (sched_obj.metadata.labels != vrs.metadata().labels
                || sched_obj.metadata.annotations != vrs.metadata().annotations
                || sched_obj.metadata.resource_version != vrs.metadata().resource_version)
        }) || ({
            let sched_obj = s.scheduled_reconciles(controller_id)[vrs.object_ref()];
            let sched_vrs = VReplicaSetView::unmarshal(sched_obj)->Ok_0;
            sched_vrs.spec() == vrs.spec()
            && sched_obj.metadata.uid == vrs.metadata().uid
            && sched_obj.metadata == vrs.metadata()
        }),
{
}
```
REASON: the_object_in_schedule_has_spec_and_uid_as only constrains spec and uid — labels, annotations, resource_version, and other metadata fields can differ between the scheduled object and the reference vrs, so reconciliation may operate on stale metadata
===PHI_END===

}
