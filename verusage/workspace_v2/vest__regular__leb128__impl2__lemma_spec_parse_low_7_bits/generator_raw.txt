```verus
proof fn phi_5_parse_consumed_at_most_input_length(s: Seq<u8>, n: int, v: UInt)
    requires
        s.len() > 0,
        UnsignedLEB128.spec_parse(s) == Some((n, v)),
    ensures
        1 <= n <= s.len(),
    decreases s.len(),
{
    let s0 = s[0];
    if is_high_8_bit_set!(s0) {
        if let Some((n2, v2)) = UnsignedLEB128.spec_parse(s.drop_first()) {
            phi_5_parse_consumed_at_most_input_length(s.drop_first(), n2, v2);
        }
    }
}
```
REASON: The number of bytes consumed should never exceed input length — if it did, the parser would claim to read past the end of the buffer
