        s3 is Send,
    ensures
        ({
            let er = EventResults {
                recvs: Seq::<NetEvent>::empty(),
                clocks: Seq::<NetEvent>::empty(),
                sends: seq![s1, s2, s3],
                ios: seq![s1, s2, s3],
            };
            er.well_typed_events()
        }),
{
}
```
REASON: `well_typed_events` bounds `clocks.len() <= 1` but places no upper bound on `sends` or `recvs` — a single protocol step can claim an unbounded number of sends, potentially allowing a single transition to flood the network

proof fn phi_4_event_seq_ordering_fixed(r: NetEvent, c: NetEvent, s: NetEvent)
    requires
        r is Receive,
        c is ReadClock,
        s is Send,
    ensures
        ({
            let er = EventResults {
                recvs: seq![r],
                clocks: seq![c],
                sends: seq![s],
                ios: seq![r, c, s],
            };
            er.event_seq() =~= seq![r, c, s]
        }),
{
}
```
REASON: `event_seq()` concatenates in fixed order `recvs + clocks + sends` — if the actual network interleaving was send-then-receive, the event sequence misrepresents the real causal order, potentially allowing invalid refinement proofs
