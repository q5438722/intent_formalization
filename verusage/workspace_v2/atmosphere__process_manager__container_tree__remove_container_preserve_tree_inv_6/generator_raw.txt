```verus
proof fn phi_1_owned_threads_shared_across_containers(
    root_container: ContainerPtr,
    container_perms: Map<ContainerPtr, PointsTo<Container>>,
    c1: ContainerPtr,
    c2: ContainerPtr,
    t: ThreadPtr,
)
    requires
        container_perms_wf(container_perms),
        container_tree_wf(root_container, container_perms),
        container_perms.dom().contains(c1),
        container_perms.dom().contains(c2),
        c1 != c2,
        container_perms[c1].value().owned_threads@.contains(t),
        container_perms[c2].value().owned_threads@.contains(t),
    ensures
        true,
{
}
```
REASON: No disjointness constraint on owned_threads across containers — two containers can claim the same thread simultaneously
