```verus
    requires
        s1.finite(),
        s2.finite(),
        s1.len() == s2.len(),
        forall |elt| s2.contains(elt) ==> s1.contains(elt),

{
    set_mismatch(s1, s2, missing);
}
```

```verus
proof fn phi_3_helper_dup1_in_range(m: Map<nat, nat>, missing: nat, len: nat)
    requires
        len >= 2,
        forall |i: nat| (0 <= i < len <==> m.dom().contains(i)),
        forall |i: nat| (#[trigger] m.dom().contains(i) ==> (
            0 <= m[i] < len && m[i] != missing
        )),
        0 <= missing < len,
    ensures ({
        let dup2 = pigeonhole_missing_idx_implies_double_helper(m, missing, len, Set::empty(), 0);
        let dup1 = choose |dup1: nat| #![auto] dup1 != dup2 && m.dom().contains(dup1) && 0 <= dup1 < len && m[dup1] == m[dup2];
        dup1 < len
    }),
{
    let dup2 = pigeonhole_missing_idx_implies_double_helper(m, missing, len, Set::empty(), 0);
}
```
REASON: The duplicate witness `dup1` is guaranteed to be `< len` by the ensures — but this relies on the `choose` selecting from the existential guaranteed by the external_body-backed proof chain
