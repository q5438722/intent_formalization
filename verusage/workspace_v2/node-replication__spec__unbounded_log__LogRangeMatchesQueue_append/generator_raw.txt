```verus
proof fn phi_4_append_preserves_existing_entries<DT: Dispatch>(
    queue: Seq<ReqId>,
    log: Map<LogIdx, LogEntry<DT>>,
    nodeId: NodeId,
    updates: Map<ReqId, UpdateState<DT>>,
    new_updates: Map<ReqId, UpdateState<DT>>,
    new_rid: ReqId,
    log_entry: LogEntry<DT>,
    logUpper: LogIdx,
)
    requires
        0 <= logUpper,
        LogContainsEntriesUpToHere(log, logUpper),
        log_entry.node_id == nodeId,
        new_updates.contains_key(new_rid),
        new_updates[new_rid] === (UpdateState::Placed { op: log_entry.op, idx: logUpper }),
        !queue.contains(new_rid),
        forall |rid| #[trigger] updates.contains_key(rid) && rid != new_rid ==> new_updates.contains_key(rid) && new_updates[rid] === updates[rid],
        LogRangeMatchesQueue(queue, log, 0, 0, logUpper, nodeId, updates),
    ensures ({
        let new_log = log.insert(logUpper, log_entry);
        let new_queue = queue.push(new_rid);
        LogRangeMatchesQueue(new_queue, new_log, 0, 0, logUpper + 1, nodeId, new_updates)
    }),
{
    LogRangeMatchesQueue_append(queue, log, log.insert(logUpper, log_entry), 0, 0, logUpper, nodeId, updates, new_updates, new_rid, log_entry);
}
```
REASON: Full instantiation of append lemma from index 0 — tests that the verified lemma correctly extends the log range by 1 with a new local entry and queue element
