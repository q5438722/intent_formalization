```verus
proof fn phi_5_max_length_lemma_individual_bound<C: SecureSpecCombinator>(c: &RepeatN<C>, vs: Seq<C::Type>, i: int)
    requires
        c.requires(),
        c.wf(vs),
        c.spec_serialize(vs).len() <= usize::MAX,
        0 <= i < vs.len(),
    ensures
        c.0.spec_serialize(vs[i]).len() <= usize::MAX,
{
    c.lemma_spec_serialize_max_length(vs, c.1);
}
```
REASON: If the total serialization fits in usize, each individual element's serialization must also fit — if any individual exceeded usize::MAX while the total didn't, the addition would have overflowed
